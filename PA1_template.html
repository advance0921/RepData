##Are there differences in activity patterns between weekdays and weekends? To compare whether activity levels are different between weekday and weekend, additional values are assigned to each data point to label the day. The plot uses the complete data with missing data filled. As we can see from the plot, the activity patterns are very different between weekdays and weekends. There are more high peaks during weekdays. It seems like the person is not as active during weekends as during weekdays. 

weekday_assign = function(datatable){
    weekday =  weekdays(as.Date(datatable$date))
    day = rep(0,length(weekday))
    for (i in 1:length(weekday)){
        if(weekday[i] %in% c("Saturday","Sunday")){
            day[i] = c("weekend")
        }
    else{
        day[i] = c("weekday")
        }
    }
    datatable$day =day
    return(datatable)
}

dt = weekday_assign(dt)
dt.all.fill = weekday_assign(dt.all.fill)
